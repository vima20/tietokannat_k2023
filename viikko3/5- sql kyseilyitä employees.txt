c) Mikä on suurin ja pienin palkka (salary) ? VIHJE: MIN, MAX

 SELECT MAX(salaries) FROM employees;
 Select MIN(salaries) FROM employees;

d) Mikä on keskimääräinen palkka ? VIHJE: AVG

SELECT ROUND(AVG(salary), 2) FROM emoloyees;

SELECT gender, COUNT(*) FROM employees GROUP BY gender;
